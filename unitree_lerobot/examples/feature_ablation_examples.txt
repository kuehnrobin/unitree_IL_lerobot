# Example configurations for feature ablation studies

# 1. Test without camera velocities and torques (positions only)
--feature_selection.use_joint_velocities=false \
--feature_selection.use_joint_torques=false \
--feature_selection.use_pressure_sensors=true

# 2. Test with only left arm (exclude right arm, hands, camera)
--feature_selection.joint_groups=["left_arm"]

# 3. Test without camera joints (useful to test if camera movement affects policy)
--feature_selection.exclude_joint_groups=["camera"]

# 4. Test with only specific cameras (e.g., only high-level view)
--feature_selection.cameras=["cam_high", "cam_low"]

# 5. Test with only stereo head cameras
--feature_selection.cameras=["cam_stereo_left", "cam_stereo_right"]

# 6. Test with single camera (minimal setup)
--feature_selection.cameras=["cam_high"]

# 7. Test without pressure sensors
--feature_selection.use_pressure_sensors=false

# 8. Test with only joint positions (no velocities, torques, or pressures)
--feature_selection.use_joint_velocities=false \
--feature_selection.use_joint_torques=false \
--feature_selection.use_pressure_sensors=false

# 9. Test excluding specific cameras (keep everything except stereo cameras)
--feature_selection.exclude_cameras=["cam_stereo_left", "cam_stereo_right"]

# 10. Complex ablation: Only arms, no hands, with velocities, minimal cameras
--feature_selection.joint_groups=["left_arm", "right_arm"] \
--feature_selection.use_joint_velocities=true \
--feature_selection.use_joint_torques=false \
--feature_selection.use_pressure_sensors=false \
--feature_selection.cameras=["cam_high"]
